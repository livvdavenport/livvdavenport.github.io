library(tidyverse)
library(gt)
library(patchwork)
library(ggrepel)

bballhits <- read_csv("mlb_hitter_stats_1914_2024.csv") |> 
  mutate(Player=gsub("#","",Player, fixed=TRUE)) |> 
  mutate(Player=gsub("*","",Player, fixed=TRUE))

bballvalue <- read_csv("mlb_hitter_value_1914_2024.csv") |> 
  mutate(Player=gsub("#","",Player, fixed=TRUE)) |> 
  mutate(Player=gsub("*","",Player, fixed=TRUE))

royalshits <- bballhits |> 
  filter(Team == "KCR") |> 
  filter(Season > 1984) |> 
  filter(grepl("\\*6", Pos) == TRUE)

royalsvalue <- bballvalue |> 
  filter(Team == "KCR") |> 
  filter(season > 1984) |> 
  filter(grepl("\\*6", Pos) == TRUE) |> 
  mutate(PlayerSeason = paste(Player, season)) 


bobby2023 <- royalshits |> 
  filter(Player == "Bobby Witt Jr.") |> 
  filter(Season == "2023")

bobby2024 <- royalshits |> 
  filter(Player == "Bobby Witt Jr.") |> 
  filter(Season == "2024")

bobbyvalue2023 <- royalsvalue |> 
  filter(Player == "Bobby Witt Jr.") |> 
  filter(season == "2023")

bobbyvalue2024 <- royalsvalue |> 
  filter(Player == "Bobby Witt Jr.") |> 
  filter(season == "2024")

bar1 <- ggplot() +
  geom_bar(data = royalsvalue, aes(x=reorder(PlayerSeason, oWAR), weight=oWAR)) +
  geom_bar(data = bobbyvalue2023, aes(x=reorder(PlayerSeason, oWAR), weight=oWAR), fill = "darkblue") +
  geom_bar(data = bobbyvalue2024, aes(x=reorder(PlayerSeason, oWAR), weight=oWAR), fill = "darkblue") +coord_flip() +
  theme_minimal() +
  labs(
    x="Player", 
    y="oWAR" 
    )

bar2 <- ggplot() +
